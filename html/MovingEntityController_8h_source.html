37 template<typename EntityModelType>
38 void MovingEntityController<EntityModelType>::update(World &world, EntityModelType &entity) {
39 float delta = Stopwatch::getInstance()->getDeltaTime();
40
41 TileMap::TileType tileInRequested = getTileInDirection(world, entity, entity.getRequestedDirection());
42
43 if (entity.getRequestedDirection() != entity.getDirection() && tileInRequested != TileMap::TileType::WALL) {
44 auto [isPastCenter, tileCenter] = _isPastOrOnCenter(world, entity, entity.getDirection());
45 auto ghost = dynamic_cast<GhostModel*>(&entity);
46 if (isPastCenter) {
47 entity.setPosition(tileCenter);
48 entity.setDirection(entity.getRequestedDirection());
49 }
50 }
51 if (entity.getDirection()==Direction::NONE) return;
52
53 TileMap::TileType tileInFront = getTileInDirection(world, entity, entity.getDirection());
54 auto [isPastCenter, tileCenter] = _isPastOrOnCenter(world, entity, entity.getDirection());
55 if (tileInFront != TileMap::TileType::WALL || !isPastCenter) {
56 if (entity.getDirection() == Direction::LEFT)
57 entity.setPosition(entity.getPosition() + Position(-entity.getSpeed(), 0)* delta);
58 else if (entity.getDirection() == Direction::RIGHT)
59 entity.setPosition(entity.getPosition() + Position(entity.getSpeed(), 0)*delta);
60 else if (entity.getDirection() == Direction::UP)
61 entity.setPosition(entity.getPosition() + Position(0, entity.getSpeed())*delta);
62 else if (entity.getDirection() == Direction::DOWN)
63 entity.setPosition(entity.getPosition() + Position(0, -entity.getSpeed())*delta);
64 }
65 else {
66 auto [isPastCenter, tileCenter] = _isPastOrOnCenter(world, entity, entity.getDirection());
67 if (isPastCenter) {
68 entity.setPosition(tileCenter);
69 onWallCollision(world, entity);
70 }
71 }

76 MovingEntityController<EntityModelType>::getTileInDirection(const World &world, const EntityModelType &entity,

78 auto [row, col] = world.getConfig().getTileMap().getGridPosition(entity.getPosition());
79
80 TileMap::TileType tileInFront;
81 switch (direction) {
82 case Direction::LEFT:
83 tileInFront = world.getConfig().getTileMap().getTileType(row, col - 1);
84 break;
85 case Direction::RIGHT:
86 tileInFront = world.getConfig().getTileMap().getTileType(row, col + 1);
87 break;
88 case Direction::UP:
89 tileInFront = world.getConfig().getTileMap().getTileType(row - 1, col);
90 break;
91 case Direction::DOWN:
92 tileInFront = world.getConfig().getTileMap().getTileType(row + 1, col);
93 break;
94 case Direction::NONE: break;
95 }

99 template<typename EntityModelType>
100 std::tuple<bool, Position>
101 MovingEntityController<EntityModelType>::_isPastOrOnCenter(const World &world, const EntityModelType &entity,
102 Direction direction) {
103 auto [row, col] = world.getConfig().getTileMap().getGridPosition(entity.getPosition());
104 Position tileCenter = world.getConfig().getTileMap().getTileCenterPosition(row, col);
105 const float epsilon = 0.0001f;
106 if (std::abs(tileCenter.getX()-entity.getPosition().getX()) < epsilon && std::abs(tileCenter.getY()-entity.getPosition().getY()) < epsilon) {
107 return {true, tileCenter};
108 }
109 switch (direction) {
110 case Direction::LEFT:
111 return {entity.getPosition().getX() < tileCenter.getX(), tileCenter};
112 case Direction::RIGHT:
113 return {entity.getPosition().getX() > tileCenter.getX(), tileCenter};
114 case Direction::UP:
115 return {entity.getPosition().getY() > tileCenter.getY(), tileCenter};
116 case Direction::DOWN:
117 return {entity.getPosition().getY() < tileCenter.getY(), tileCenter};
118 default:return {false, Position(0,0)};
119 }
120 }
